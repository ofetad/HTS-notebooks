trim_func() {
    FASTQ=$1
    FASTQ_BASE=${FASTQ##*/} # strip directory from file path
    SAMPLE="${FASTQ_BASE%_R1_001.fastq.gz}" # strip .fq.gz file extension
    echo $SAMPLE
    echo $FASTQ
    echo $FASTQ_BASE
    # exit 1

    fastq-mcf \
        $ADAPTERS \
        $FASTQ \
        -o $TRIMMED/${SAMPLE}.trim.fastq.gz \
        -q 20 -x 0.5
}
export -f trim_func
